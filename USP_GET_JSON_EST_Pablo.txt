DECLARE @FECHA DATETIME = CONVERT(DATETIME,@pFECHA_DATOS,103)
DECLARE @ALTA DATETIME = CONVERT(DATETIME,@pFECHA_ALTA,103)

DECLARE @FECHA_ALTA DATE = @ALTA;
DECLARE @SECUENCIA INT =  @pSECUENCIA;
DECLARE @GRUPO_ENTIDAD INT = @pGRUPO_ENTIDAD;

DECLARE @ID_GRUPO_ENVIO VARCHAR(10) = @pID_GRUPO_ENVIO;

declare @vT_EBEXC_JSON_AUX_RGI table (
	ID_INFORME [varchar](10) COLLATE Modern_Spanish_CI_AI NOT NULL,
	ID_SUBINFORME [varchar](20) COLLATE Modern_Spanish_CI_AI NOT NULL,
	ID_VARIACION_1 [varchar](20) COLLATE Modern_Spanish_CI_AI NOT NULL,
	ID_VARIACION_2 [varchar](20) COLLATE Modern_Spanish_CI_AI NOT NULL,
	ID_VARIACION_FILA [varchar](504) COLLATE Modern_Spanish_CI_AI,
	CODIGO_CASILLA [varchar](20) COLLATE Modern_Spanish_CI_AI,
	FORMATO [varchar](20) COLLATE Modern_Spanish_CI_AI,
	ID_CELDA [varchar](50) COLLATE Modern_Spanish_CI_AI,
	VALOR [varchar](400) COLLATE Modern_Spanish_CI_AI
	primary key (ID_CELDA, ID_INFORME, ID_SUBINFORME, ID_VARIACION_1, ID_VARIACION_2, ID_VARIACION_FILA, CODIGO_CASILLA)
) 

BEGIN

	-- Obtencion CELDAS del fiichero
	set @vPaso = 1;
	/*
	--select rgef.CLAVE_OFICIAL_INFORME, rgef.COD_PLANTILLA_JSON, rgef.ES_OPCIONAL, rgef.TIPO_DATO_MOTOR_EXP, rgef.ORDEN, ef.VARIACION, ef.NOMBRE_ESTRUCTURA, ef.UNIQUE_KEYS

	select
		c.CLAVE_OFICIAL_INFORME, c.ID_CELDA, c.ID_FICHERO, c.NOMBRE_ESTRUCTURA, c.NOMBRE_CAMPO_FICHERO, c.DONDE_ESTA_EL_VALOR, c.ID_TIPO_DATO, c.DECIMALES, c.UNIDADES
	from T_EBEX_GRUPO_ENVIO ge
		INNER JOIN T_EBEXC_REL_GRUPO_ENVIO_FICH rgef
			ON (ge.ID_GRUPO_ENVIO = rgef.ID_GRUPO_ENVIO AND ge.FECHA_ALTA = rgef.FECHA_ALTA)
		inner join T_EBEXC_CELDA c
			on (c.ID_GRUPO_ENVIO = rgef.ID_GRUPO_ENVIO AND ge.FECHA_ALTA = rgef.FECHA_ALTA and c.CLAVE_OFICIAL_INFORME = rgef.CLAVE_OFICIAL_INFORME and c.ID_FICHERO = rgef.ID_FICHERO)
	where 1=1
		AND ge.ID_GRUPO_ENVIO = @ID_GRUPO_ENVIO 
		AND ge.FECHA_ALTA = @FECHA_ALTA
		AND ge.ES_XBRL IN (5)
	order by c.CLAVE_OFICIAL_INFORME, c.ID_CELDA
	;
	*/

	-- Obtención Valores del estado
	set @vPaso = 10;

	INSERT INTO @vT_EBEXC_JSON_AUX_RGI(ID_INFORME, ID_SUBINFORME, ID_VARIACION_1, ID_VARIACION_2, ID_VARIACION_FILA, CODIGO_CASILLA, FORMATO, ID_CELDA, VALOR)
	select  rgi.ID_INFORME, rgi.ID_SUBINFORME, rgi.ID_VARIACION_1, rgi.ID_VARIACION_2, rgi.ID_VARIACION_FILA, rgi.CODIGO_CASILLA, cs.FORMATO, cs.ID_CELDA, rgi.valor AS VALOR
		from t_ebe_resumen_generacion_infor rgi
		inner join T_EBE_CASILLA_SUBINFORME cs 
			on (rgi.ID_SUBINFORME = cs.ID_SUBINFORME and rgi.ID_CASILLA = cs.ID_CASILLA)
	where 1=1
		and rgi.ID_GRUPO_ENTIDAD = @GRUPO_ENTIDAD
		and rgi.ID_INFORME = @ID_INFORME
		and rgi.SEC_INFORME_GENERADO = @SECUENCIA
		and rgi.FECHA_DATOS_INFORME_GENERADO = @FECHA
		and cs.ID_TIPO_CASILLA = 'DATO'
	order by rgi.ID_INFORME, rgi.ID_SUBINFORME, rgi.ID_VARIACION_1, rgi.ID_VARIACION_2, rgi.ID_VARIACION_FILA, cs.ID_CELDA
	;


	select
		c.CLAVE_OFICIAL_INFORME, c.ID_CELDA, c.ID_FICHERO, c.NOMBRE_ESTRUCTURA, c.NOMBRE_CAMPO_FICHERO, c.DONDE_ESTA_EL_VALOR, c.ID_TIPO_DATO, c.DECIMALES, c.UNIDADES
		,auxrgi.ID_INFORME, auxrgi.ID_SUBINFORME, auxrgi.ID_VARIACION_1, auxrgi.ID_VARIACION_2, auxrgi.ID_VARIACION_FILA, auxrgi.CODIGO_CASILLA, auxrgi.FORMATO, auxrgi.VALOR, c.USOS_DE_NULL
	from T_EBEX_GRUPO_ENVIO ge
		INNER JOIN T_EBEXC_REL_GRUPO_ENVIO_FICH rgef
			ON (ge.ID_GRUPO_ENVIO = rgef.ID_GRUPO_ENVIO AND ge.FECHA_ALTA = rgef.FECHA_ALTA)
		inner join T_EBEXC_CELDA c
			on (c.ID_GRUPO_ENVIO = rgef.ID_GRUPO_ENVIO AND c.FECHA_ALTA = rgef.FECHA_ALTA and c.CLAVE_OFICIAL_INFORME = rgef.CLAVE_OFICIAL_INFORME and c.ID_FICHERO = rgef.ID_FICHERO)
		left join @vT_EBEXC_JSON_AUX_RGI auxrgi 
			on (c.ID_CELDA = auxrgi.ID_CELDA)
	where 1=1
		AND ge.ID_GRUPO_ENVIO = @ID_GRUPO_ENVIO 
		AND ge.FECHA_ALTA = @FECHA_ALTA
		AND ge.ES_XBRL IN (5)
		AND rgef.CLAVE_OFICIAL_INFORME = @vID_ESTADO
	order by c.CLAVE_OFICIAL_INFORME, auxrgi.ID_VARIACION_1, auxrgi.ID_VARIACION_2, auxrgi.ID_VARIACION_FILA, c.ID_CELDA;

	
	set @vPaso = 20;

END
